async function searchStock() { const input = document.getElementById('stockInput').value.trim(); const dashboard = document.getElementById('dashboard'); if (!input) { alert("請輸入股票代號！"); return; } // 顯示「載入中」的感覺（選做：可以用 alert 簡單提示，或直接執行 fetch） console.log("正在分析：" + input); try { // 這行是關鍵！它會去呼叫你剛建好的 Netlify 後端 const API_URL = `/.netlify/functions/stockInfo?id=${input}`; const response = await fetch(API_URL); const data = await response.json(); if (data.error) { alert("錯誤: " + data.error); return; } // --- 把抓到的真實資料填入網頁 --- document.getElementById('stockName').innerText = `${data.id} ${data.name}`; document.getElementById('stockPe').innerText = data.pe + " 倍"; document.getElementById('stockYield').innerText = data.yield + " %"; // 額外資訊：股價淨值比 document.getElementById('debtRatio').innerText = data.pb; document.getElementById('grossMargin').innerText = "官方 API 提供中"; // 自動化評估邏輯 let summary = ""; const pe = parseFloat(data.pe); if (pe < 15) { summary = "目前本益比低於 15 倍，估值相對便宜。建議檢查基本面是否穩健。"; document.getElementById('healthScore').innerText = "良 (🌟🌟🌟🌟)"; } else if (pe > 25) { summary = "本益比高於 25 倍，市場預期較高。建議分批佈局，不宜追高。"; document.getElementById('healthScore').innerText = "普 (🌟🌟🌟)"; } else { summary = "本益比處於合理區間，適合長期追蹤營收成長。"; document.getElementById('healthScore').innerText = "優 (🌟🌟🌟🌟🌟)"; } document.getElementById('summaryText').innerText = summary; document.getElementById('buyPrice').innerText = "請參考本益比區間"; document.getElementById('sellPrice').innerText = "視技術面壓力而定"; // 顯示結果 dashboard.style.display = 'grid'; } catch (err) { alert("連線失敗，請確定後端已部署成功！"); console.error(err); } }
